# 一、新语法

# 1. 类型延迟标注

3.7 之前的版本，类型标注只能使用在当前作用域中已经存在的名称，也就是说，它们不支持任何形式的前向引用 ，3.7 则可以实现向前引用。 注：它将在 Python4.0 中成为默认行为。

from __future__ import annotations


class C:
    @classmethod
    def from_string(cls, source: str) -> C:
        ...

    def validate_b(self, obj: B) -> bool:
        ...


class B:
    ...

# 二、新的库模块

# 1. contextvars

上下文变量

未使用上下文管理器：

import asyncio

request_id = None


async def get():
    # Get Value
    print(f'Request ID (Inner): {request_id}')


async def new_coro(req_id):
    # Set Value
    request_id = req_id
    await get()
    print(f'Request ID (Outer): {request_id}')


async def main():
    tasks = []
    for req_id in range(1, 5):
        tasks.append(asyncio.create_task(new_coro(req_id)))

    await asyncio.gather(*tasks)


if __name__ == '__main__':
    asyncio.run(main())

使用上下文管理器：

import asyncio
import contextvars

# 申明Context变量
request_id = contextvars.ContextVar('Id of request')


async def get():
    # Get Value
    print(f'Request ID (Inner): {request_id.get()}')


async def new_coro(req_id):
    # Set Value
    request_id.set(req_id)
    await get()
    print(f'Request ID (Outer): {request_id.get()}')


async def main():
    tasks = []
    for req_id in range(1, 5):
        tasks.append(asyncio.create_task(new_coro(req_id)))

    await asyncio.gather(*tasks)


if __name__ == '__main__':
    asyncio.run(main())

# 2. dataclasses

数据类

@dataclass
class Point:
    x: float
    y: float
    z: float = 0.0

p = Point(1.5, 2.5)
print(p)   # produces "Point(x=1.5, y=2.5, z=0.0)"

# 三、新特性

# 1. 字典有序

3.7 中的字典变得有序。

# 2. 标准库time支持纳秒级

在原有的基础上新增以下 6 个纳秒级函数。

  • time.clock_gettime_ns()

  • time.clock_settime_ns()

  • time.monotonic_ns()

  • time.perf_counter_ns()

  • time.process_time_ns()

  • time.time_ns()

# 3. 文本编码

强制使用 UTF-8 用作默认的文本编码。
